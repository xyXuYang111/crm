java反射机制完成List<map>与List<javabean>,List<对象>之间的互相转换
2019-03-06 00:09 许多年以后 1492

我们可以利用Java的反射机制list<map>与list<对象>的就是list< avabean>之间的互转，可以参考下面代码:

编写工具类

  1. public class ListUtil{
  2. /**
  3.      * 将JavaBean转换成Map
  4.      * 
  5.      * @param obj
  6.      * @return
  7.      * @throws SecurityException
  8.      * @throws NoSuchMethodException
  9.      * @throws InvocationTargetException
  10.      * @throws IllegalArgumentException
  11.      * @throws IllegalAccessException
  12.      */
  13.     public static Map beanToMap(Object obj) throws NoSuchMethodException, SecurityException, IllegalAccessException,
  14.             IllegalArgumentException, InvocationTargetException {
  15.         // 创建map集合
  16.         Map map = new HashMap();
  17.         // 获取JavaBean中所有属性
  18.         Field[] fields = obj.getClass().getDeclaredFields();
  20.         for (Field fie : fields) {
  22.             // 将属性第一个字母转换成大写
  23.             String frist = fie.getName().substring(0, 1).toUpperCase();
  24.             // 获取属性的类型
  25.             Class<?> type = fie.getType();
  26.             // 封装属性的get
  27.             String getter = "";
  28.             if ("boolean".equals(type.getName())) {
  29.                 getter = "is" + frist + fie.getName().substring(1);
  30.             } else {
  31.                 getter = "get" + frist + fie.getName().substring(1);
  32.             }
  33.             // 获取JavaBean的方法
  34.             Method method = obj.getClass().getMethod(getter, new Class[] {});
  35.             // 调用方法,并接收返回值
  36.             Object objec = method.invoke(obj, new Object[] {});
  38.             // 判断返回值不为空
  39.             if (objec != null) {
  40.                 map.put(fie.getName(), objec);
  41.             } else {
  42.                 map.put(fie.getName(), "");
  43.             }
  44.         }
  46.         return map;
  47.     }
  49.     /**
  50.      * 将Map转换为JavaBean
  51.      * 
  52.      * @param map
  53.      * @param obj
  54.      * @return
  55.      * @throws SecurityException
  56.      * @throws NoSuchMethodException
  57.      * @throws InvocationTargetException
  58.      * @throws IllegalArgumentException
  59.      * @throws IllegalAccessException
  60.      */
  61.     public static Object mapToBean(Map<String, Object> map, Object obj) throws NoSuchMethodException, SecurityException,
  62.             IllegalAccessException, IllegalArgumentException, InvocationTargetException {
  64.         // 获取JavaBean中的所有属性
  65.         Field[] field = obj.getClass().getDeclaredFields();
  67.         for (Field fi : field) {
  68.             // 判断key值是否存在
  69.             if (map.containsKey(fi.getName())) {
  70.                 // 获取key的value值
  71.                 String value = map.get(fi.getName()).toString();
  72.                 // 将属性的第一个字母转换为大写
  73.                 String frist = fi.getName().substring(0, 1).toUpperCase();
  74.                 // 属性封装set方法
  75.                 String setter = "set" + frist + fi.getName().substring(1);
  76.                 // 获取当前属性类型
  77.                 Class<?> type = fi.getType();
  78.                 // 获取JavaBean的方法,并设置类型
  79.                 Method method = obj.getClass().getMethod(setter, type);
  81.                 // 判断属性为double类型
  82.                 if ("java.lang.String".equals(type.getName())) {
  84.                   // 调用当前Javabean中set方法，并传入指定类型参数
  85.                     method.invoke(obj, value);
  87.                 } else if ("int".equals(type.getName())) {
  89.                     method.invoke(obj, Integer.parseInt(value));
  91.                 }else if ("double".equals(type.getName())) {
  93.                     method.invoke(obj, Double.valueOf(value));
  94.                  } else if ("char".equals(type.getName())) {
  96.                     method.invoke(obj, value);
  98.                 } 
  99.             }
  100.         }
  102.         return obj;
  103.     }
  105.     /**
  106.      * 将List<Map<String,Object>>转换成List<javaBean>
  107.      * 
  108.      * @param listm
  109.      * @param obj
  110.      * @return
  111.      * @throws InvocationTargetException
  112.      * @throws IllegalArgumentException
  113.      * @throws IllegalAccessException
  114.      * @throws SecurityException
  115.      * @throws NoSuchMethodException
  116.      */
  117.     public static Object ListMapToListBean(List<Map<String, Object>> listm, Object obj)
  118.             throws NoSuchMethodException, SecurityException, IllegalAccessException, IllegalArgumentException,
  119.             InvocationTargetException {
  121.         List<Object> list = new ArrayList<Object>();
  122.         // 循环遍历出map对象
  123.         for (Map<String, Object> m : listm) {
  124.             // 调用将map转换为JavaBean的方法
  125.             Object objs = mapToBean(m, obj);
  126.             // 添加进list集合
  127.             list.add(objs);
  128.         }
  130.         return list;
  131.     }
  133.     /**
  134.      * 将list<javabean>转换为List<Map>
  135.      * 
  136.      * @param list
  137.      * @return
  138.      * @throws NoSuchMethodException
  139.      * @throws SecurityException
  140.      * @throws IllegalAccessException
  141.      * @throws IllegalArgumentException
  142.      * @throws InvocationTargetException
  143.      */
  144.     public static List<Map<String, Object>> ListBeanToListMap(List<Object> list) throws NoSuchMethodException,
  145.             SecurityException, IllegalAccessException, IllegalArgumentException, InvocationTargetException {
  147.         List<Map<String, Object>> listmap = new ArrayList<Map<String, Object>>();
  149.         for (Object ob : list) {
  151.             listmap.add(beanToMap(ob));
  152.         }
  154.         return listmap;
  155.     }
  156. }

用法：

  1. public class Test{
  2. private String name;
  3. //setting和gentting
  4. }
  6. //将List<Map<String,Object>>转换成List<javaBean>
  7. List<Map<String, Object>> list=new ArrayList<Map<String, Object>>();
  8. Map<String, Object> map=new HashMap<String, Object>();
  9. map.add("name","shusheng");
  10. list.add(map);
  12. List<Test> test=ListUtil.ListMapToListBean(list,new Test());
