使用Hyperic-Sigar获取服务器设备信息（监控服务器）
2019-09-09 09:26 许多年以后 1449

有时候我们要监控服务器各种设备信息，那么想要用java语言开发时，可以用Hyperic-Sigar。
Hyperic-Sigar是一个收集系统各项底层信息的工具集.他有如下特点:

    1. 收集信息全面
  • 收集CPU,MEM,NETWORK,PROCESS,IOSTAT等
  • 使用Sigar,你完全可以模仿出cpuinfo,meminfo,top,free,ifconfig,ipconfig,netstat,route,df,du,ps,ls等多种unix平台和windows平台的指令.
  • 2.跨平台,支持多数平台
  • 支持的平台包括:windows系列(32系列,IA64系列,AMD64系列),Linux系列,freeBsd系列,HPUnix系列,Sun solaris/Sparc/Sparc64系列,macOs系列,AIX系列等。

  237.     public String getDefaultIpAddress() { 
  238.         String address = null; 
  239.         try { 
  240.             address = InetAddress.getLocalHost().getHostAddress(); 
  241.             // 没有出现异常而正常当取到的IP时，如果取到的不是网卡循回地址时就返回 
  242.             // 否则再通过Sigar工具包中的方法来获取 
  243.             if (!NetFlags.LOOPBACK_ADDRESS.equals(address)) { 
  244.                 return address; 
  245.             } 
  246.         } catch (UnknownHostException e) { 
  247.             // hostname not in DNS or /etc/hosts 
  248.         } 
  249.         Sigar sigar = new Sigar(); 
  250.         try { 
  251.             address = sigar.getNetInterfaceConfig().getAddress(); 
  252.         } catch (SigarException e) { 
  253.             address = NetFlags.LOOPBACK_ADDRESS; 
  254.         } finally { 
  255.             sigar.close(); 
  256.         } 
  257.         return address; 
  258.     } // c)取到当前机器的MAC地址 
  259.     public String getMAC() { 
  260.         Sigar sigar = null; 
  261.         try { 
  262.             sigar = new Sigar(); 
  263.             String[] ifaces = sigar.getNetInterfaceList(); 
  264.             String hwaddr = null; 
  265.             for (int i = 0; i < ifaces.length; i++) { 
  266.                 NetInterfaceConfig cfg = sigar.getNetInterfaceConfig(ifaces[i]); 
  267.                 if (NetFlags.LOOPBACK_ADDRESS.equals(cfg.getAddress()) || (cfg.getFlags() & NetFlags.IFF_LOOPBACK) != 0 || NetFlags.NULL_HWADDR.equals(cfg.getHwaddr())) { 
  268.                     continue; 
  269.                 } 
  270.       /* 
  271.        * 如果存在多张网卡包括虚拟机的网卡，默认只取第一张网卡的MAC地址，如果要返回所有的网卡（包括物理的和虚拟的）则可以修改方法的返回类型为数组或Collection 
  272.        * ，通过在for循环里取到的多个MAC地址。 
  273.        */
  274.                 hwaddr = cfg.getHwaddr(); 
  275.                 break; 
  276.             } 
  277.             return hwaddr != null?hwaddr : null; 
  278.         } catch (Exception e) { 
  279.             return null; 
  280.         } finally { 
  281.             if (sigar != null) 
  282.                 sigar.close(); 
  283.         } 
  284.     }
